33 def otsu_global(matrix: np.ndarray) -> tuple: 

34 """ 

35 Глобальная бинаризация методом Отсу 

36 

37 :param matrix: матрица изображения 

38 :type numpy.ndarray 

39 

40 :return: tuple -- image threshold, black mean, white mean 

41 """ 

42 n_curr = 0 

43 T_res = 0 

44 M0_res = 0 

45 M1_res = 0 

46 

47 p_tmp = np.unique(matrix, return_counts=True) 

48 p = p_tmp[1] / matrix.size 

49 

50 for t in range(matrix.min(), matrix.max()): 

51 w0 = p[p_tmp[0] <= t].sum() if p[p_tmp[0] <= t].sum() > 0.00001 else 0.00001 

52 w1 = 1 - w0 if 1 - w0 > 0.00001 else 0.00001 

53 M0 = (p_tmp[0][p_tmp[0] <= t] * p[p_tmp[0] <= t]).sum() / w0 

54 M1 = (p_tmp[0][p_tmp[0] > t] * p[p_tmp[0] > t]).sum() / w1 

55 D0 = (p[p_tmp[0] <= t] * np.square(p_tmp[0][p_tmp[0] <= t] - M0)).sum() 

56 D1 = (p[p_tmp[0] > t] * np.square(p_tmp[0][p_tmp[0] > t] - M1)).sum() 

57 

58 n = (w0 * w1 * (M0 - M1)**2) // (w0 * D0 + w1 * D1) 

59 if n >= n_curr: 

60 n_curr = n 

61 T_res = t 

62 M0_res = M0 

63 M1_res = M1 

64 

65 return T_res, M0_res, M1_res

68 def split_submatrix(matrix: np.ndarray, submat1_shape: tuple, submat2_shape: tuple): 

69 """ 

70 Генерирует кортеж из начальных и конечных координат для малого и большего окна 

71 

72 :param matrix: матрица изображения 

73 :type numpy.ndarray 

74 :param submat1_shape: размер малого окна 

75 :type submat1_shape: tuple 

76 :param submat2_shape: размер большего окна 

77 :type submat2_shape: tuple 

78 """ 

79 p, q = submat1_shape 

80 P, Q = submat2_shape 

81 m, n = matrix.shape 

82 

83 bias_p = (P - p) // 2 

84 bias_q = (Q - q) // 2 

85 for x in range(0, m, p): 

86 for y in range(0, n, q): 

87 yield ( 

88 ( 

89 (x, (x + p) if (x + p - m) < 0 else m), 

90 (y, (y + q) if (y + q - n) < 0 else n) 

91 ), 

92 ( 

93 ((x - bias_p) if (x - bias_p) > 0 else 0, (x + P - bias_p) if (x + P - bias_p) < m else m), 

94 ((y - bias_q) if (y - bias_q) > 0 else 0, (y + Q - bias_q) if (y + Q - bias_q) < n else n) 

95 )

107 matrix_k_ind[1][0]: matrix_k_ind[1][1]]

109 matrix_K_ind[1][0]: matrix_K_ind[1][1]] 

110 T, M0, M1 = otsu_global(matrix_K) 

111

115 else:

117 new_T = (M0 + M1) / 2 

118 self.bin_matrix[matrix_k_ind[0][0]: matrix_k_ind[0][1], matrix_k_ind[1][0]: matrix_k_ind[1][1]]\

134 def calc_integ(self, img): 

135 """ 

136 Расчет интегрального изображения из исходного 

137 

138 :param img: матрица изображения 

139 :type img: numpy.ndarray 

140 """ 

141 h, w = img.shape 

142 integr = np.zeros((h, w), np.float) 

143 #сначала посчитаем для первого столбца и первой строки, чтоб не выходить за границы в рекурсивной формуле 

144 integr[0, :] = np.cumsum(img[0, :]) 

145 integr[:, 0] = np.cumsum(img[:, 0]) 

146 #рекурсивная формула, 20 слайд 

147 for y in range(1,h): 

148 for x in range(1,w): 

149 integr[y,x] = img[y,x] - integr[y-1,x-1] + integr[y-1,x] + integr[y,x-1] 

150 return integr

153 def cristian_binarisation(self, w_size=15, k=0.5): 

154 """ 

155 Бинаризация изображения методом Кристиана 

156 

157 :param w_size: размер окна 

158 :type w_size: int 

159 :param k: коэффициент, отвечающий за чувствительность бинаризатора 

160 :type k: int 

161 """ 

162 if self.grayscale_matrix is None: 

163 self.calc_grayscale_matrix() 

164 rows, cols = self.grayscale_matrix.shape 

165 pix = self.grayscale_matrix 

166 

167 integr = self.calc_integ(pix) 

168 npix = pix.astype(int) 

169 n = np.square(npix) 

170 sqr_integr = self.calc_integ(np.square(npix)) 

171 

172 half_w = w_size // 2 

173 

174 #сейчас будут магические вещи: сделаем массив, в котором для каждого пикселя будут индексы границы окна 

175 #я сам немного удивлен, что это работает, но это работает!  

176 x, y = np.meshgrid(np.arange(0, cols), np.arange(0, rows)) 

177 

178 x1 = (x - half_w).clip(0, cols-1) #левая граница по x 

179 x2 = (x + half_w).clip(0, cols-1) #правая граница по х 

180 y1 = (y - half_w).clip(0, rows-1) #левая граница по y 

181 y2 = (y + half_w).clip(0, rows-1) #правая граница по y 

182 

183 # площадь окна для каждого пикселя. Будет другая у граничных, поэтому это тут 

184 s = (y2 - y1 + 1) * (x2 - x1 + 1) 

185 

186 

187 sums = np.zeros((rows, cols), np.float) 

188 for y in range(0,rows): 

189 for x in range(0,cols): 

190 sums[y,x] = integr[y2[y,x], x2[y,x]] - integr[y2[y,x], x1[y,x]] - integr[y1[y,x], x2[y,x]] + integr[y1[y,x] , x1[y,x]] # 0_O 

191 

192 means = sums / s 

193 

194 

195 dev_sums = np.zeros((rows, cols), np.float) 

196 for y in range(0,rows): 

197 for x in range(0,cols): 

198 dev_sums[y,x] = sqr_integr[y2[y,x], x2[y,x]] - sqr_integr[y2[y,x], x1[y,x]] - sqr_integr[y1[y,x], x2[y,x]] + sqr_integr[y1[y,x] , x1[y,x]] # 0_O [2] 

199 devs = np.sqrt(dev_sums / s - np.square(means)) 

200 

201 

202 R = np.max(devs) 

203 M = np.min(self.grayscale_matrix) 

204 

205 

206 thresholds = ((1.0 - k) * means + k * M + k * devs / R * (means - M)) 

207 img = ((self.grayscale_matrix >= thresholds) * 255).astype(np.uint8) #255, если больше, 0 если меньше 

208 self.result = Image.fromarray(np.uint8(img) , 'L') 

209 self.bin_matrix = img 

210 return self

213def otsu(image): 

214 """ 

215 Метод Отсу 

216 

217 :param image: изображение 

218 :type PIL.Image 

219 

220 :return: int -- порог бинаризации 

221 """ 

222 hist = cv2.calcHist([np.asarray(image)], [0], None, [256], [0, 256]) 

223 bins = np.arange(256) 

224 hist_norm = hist.ravel() / hist.max() 

225 Q = np.cumsum(hist_norm) 

226 fn_min = np.inf 

227 thresh = -1 

228 for i in list(range(1, 256)): 

229 p1, p2 = np.hsplit(hist_norm, [i]) # probabilities 

230 q1, q2 = Q[i], Q[255] - Q[i] # cum sum of classes 

231 if q1 == 0: 

232 q1 = 0.00000001 

233 if q2 == 0: 

234 q2 = 0.00000001 

235 b1, b2 = np.hsplit(bins, [i]) # weights 

236 # finding means and variances 

237 m1, m2 = np.sum(p1 * b1) / q1, np.sum(p2 * b2) / q2 

238 v1, v2 = np.sum(((b1 - m1) ** 2) * p1) / q1, np.sum(((b2 - m2) ** 2) * p2) / q2 

239 # calculates the minimization function 

240 fn = v1 * q1 + v2 * q2 

241 if fn < fn_min: 

242 fn_min = fn 

243 thresh = i 

244 res_m1, res_m2 = m1, m2 

245 return thresh
